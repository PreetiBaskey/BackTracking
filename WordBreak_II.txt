class Solution {
public:
    void backtrack(int i, string s, unordered_set<string> &st, string &currSentence, vector<string> &result) {
        if(i == s.length()) {
            result.push_back(currSentence);
            return;
        }
        for(int j = i; j < s.length(); j++) {
            string tempWord = s.substr(i, j + 1 - i);
            if(st.count(tempWord)) {
                string tempSentence = currSentence;
                if(!currSentence.empty()) {
                    currSentence += " ";
                }
                currSentence += tempWord;
                backtrack(j + 1, s, st, currSentence, result);
                currSentence = tempSentence;
            }
        }
    }
    vector<string> wordBreak(string s, vector<string>& wordDict) {
        vector<string> result;
        string currSentence = "";
        int i = 0;
        unordered_set<string> st;
        for(string &word: wordDict) {
            st.insert(word);
        }

        backtrack(i, s, st, currSentence, result);

        return result;
    }
};
