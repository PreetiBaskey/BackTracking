class Solution {
public:
    int n;
    void dfs(string &tiles, vector<bool> &used, unordered_set<string> &result, string &curr) {
        result.insert(curr);

        for(int i = 0; i < tiles.length(); i++) {
            if(used[i]) {
                continue;
            }
            else {
                used[i] = true;
                curr.push_back(tiles[i]);

                dfs(tiles, used, result, curr);

                used[i] = false;
                curr.pop_back();
            }
        }
    }
    int numTilePossibilities(string tiles) {
        n = tiles.length();
        vector<bool> used(n, false);
        unordered_set<string> result;
        string curr = "";

        dfs(tiles, used, result, curr);

        return result.size() - 1;
    }
};
