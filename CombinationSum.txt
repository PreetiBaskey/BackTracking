class Solution {
public:
    void backtrack(vector<int> &candidates, vector<vector<int>> &result, vector<int> &curr, int target, int idx) {
        if(target < 0) {
            return;
        }
        if(target == 0) {
            result.push_back(curr);
        }
        for(int i = idx; i < candidates.size(); i++) {
            curr.push_back(candidates[i]);
            backtrack(candidates, result, curr, target - candidates[i], i);
            curr.pop_back();
        }
    }
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        vector<vector<int>> result;
        int idx = 0;
        vector<int> curr;

        backtrack(candidates, result, curr, target, idx);

        return result;
    }
};
