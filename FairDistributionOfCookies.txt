class Solution {
public:
    int n;
    int result = INT_MAX;
    void solve(vector<int> &cookies, vector<int> &children, int k, int idx) {
        if(idx >= n) {
            int unfairness = *max_element(begin(children), end(children));
            result = min(result, unfairness);

            return;
        }

        int cookie = cookies[idx];
        for(int i = 0; i < k; i++) {
            children[i] += cookie;

            solve(cookies, children, k, idx + 1);

            children[i] -= cookie;
        }
    }
    int distributeCookies(vector<int>& cookies, int k) {
        n = cookies.size();
        vector<int> children(k, 0);

        solve(cookies, children, k, 0);

        return result;
    }
};
